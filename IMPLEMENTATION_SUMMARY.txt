BASE DE DATOS - TABLA USERS
================================================================================

CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT NOT NULL UNIQUE,
  name TEXT,
  image TEXT,
  google_id TEXT,
  preferred_language TEXT DEFAULT 'es',
  currency TEXT DEFAULT 'USD',
  timezone TEXT DEFAULT 'America/Mexico_City',
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

ÍNDICES:
- idx_users_email
- idx_users_google_id

================================================================================
API ENDPOINTS
================================================================================

GET /api/user/profile
- Requiere: autenticación NextAuth
- Retorna: perfil completo del usuario
- Maneja: 401 (unauthorized), 404 (not found), 500 (error)

PATCH /api/user/profile
- Requiere: autenticación NextAuth
- Body: { name?, preferred_language?, currency?, timezone? }
- Valida: tipos, valores permitidos
- Retorna: perfil actualizado
- Maneja: 400 (invalid), 401 (unauthorized), 404 (not found), 500 (error)
